\begin{lemma}
    $\forall n \ge 2, \forall 2 \le i < n, \gcd(n, i) = \gcd(n, n - i)$
\end{lemma}

\begin{proof}
不妨设$i < \frac{n}{2}$, 则$n \mod (n - i) = n - (n - i) = i$.
又, 令$n = pi + q (q < i)$则:\begin{align*}
    \gcd(n, i) &= \gcd(q, i) \\
    \gcd(n, n - i) &= \gcd(i, (p - 1)i + q)
\end{align*}

明显, 等号右边两式相等.
\end{proof}
